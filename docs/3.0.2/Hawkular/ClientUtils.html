def hawk_escape(url_part)
  return url_part.to_s if url_part.is_a?(Numeric)

  url_part
    .to_s
    .dup
    .gsub('%', '%25')
    .gsub(' ', '%20')
    .gsub('[', '%5b')
    .gsub(']', '%5d')
    .gsub('|', '%7c')
    .gsub('(', '%28')
    .gsub(')', '%29')
    .gsub('/', '%2f')
end

def hawk_escape_id(url_part)
  hawk_escape(url_part)
    .gsub('=', '%3d')
    .gsub(';', '%3b')
end
